CREATE DATABASE soft_travel;
USE soft_travel;

CREATE TABLE divisions(
    id int NOT NULL PRIMARY KEY AUTO_INCREMENT,
    name varchar(10) NOT NULL UNIQUE
);

CREATE TABLE spots(
    id int NOT NULL PRIMARY KEY AUTO_INCREMENT,
    name varchar(20) NOT NULL UNIQUE
);

CREATE TABLE hotels(
    id int NOT NULL PRIMARY KEY AUTO_INCREMENT,
    name varchar(50) NOT NULL,
    area varchar(10)
);
